BUTTON SEARCH ohne Filter
alle Compounds mit smiles und common name

        PREFIX onto: <http://nubbekg.aksw.org/ontology#>
        PREFIX data: <http://nubbekg.aksw.org/data#>

        SELECT ?compound ?commonName ?smiles
        WHERE {
        ?compound a onto:Compound .
        OPTIONAL { ?compound onto:commonName ?commonName }
        OPTIONAL { ?compound onto:smiles ?smiles }
        }


Biological Properties für Dropdown

        PREFIX onto: <http://nubbekg.aksw.org/ontology#>
        PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

        SELECT ?label
        WHERE {
        ?bioactivity a onto:Bioactivity .
        OPTIONAL { ?bioactivity rdfs:label ?label }
        }


States für Sample Location

        PREFIX onto: <http://nubbekg.aksw.org/ontology#>
        PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

        SELECT ?label
        WHERE {
        ?state a onto:State .
        OPTIONAL { ?state rdfs:label ?label }
        }
